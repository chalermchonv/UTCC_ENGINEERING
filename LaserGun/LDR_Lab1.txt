const int ledPin = 13;
const int ldrPin = A0;

void setup()
{   
  Serial.begin(9600);
  pinMode(ledPin, OUTPUT);
  pinMode(ldrPin, INPUT);
}

void loop()
{   
  int ldr = analogRead(ldrPin);
  Serial.println(ldr);  
  
  if (ldr > 300) digitalWrite(ledPin, HIGH);
  else digitalWrite(ledPin, LOW);

  delay(1000);

}
